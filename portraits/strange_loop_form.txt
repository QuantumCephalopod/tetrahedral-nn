#!/usr/bin/env python3
"""
Painting the Strange Loop - pure visual form
No labels. Just geometry.
"""
import math

W = 100
H = 60

# Initialize canvas
canvas = [[' ' for _ in range(W)] for _ in range(H)]

def paint_flow(canvas, cx, cy, closed=True):
    """Paint a gradient flow - closed loop or broken"""
    for i in range(H):
        for j in range(W):
            # Parametric circle/loop
            t = (i / H) * 2 * math.pi
            r = 15 + 3 * math.sin(3 * t)  # Varying radius

            x = cx + r * math.cos(t)
            y = cy + r * math.sin(t) * 0.6  # Elliptical

            dist = math.sqrt((j - x)**2 + (i - y)**2)

            # If broken, skip the gap
            if not closed and abs(t - math.pi) < 0.7:
                continue

            # Paint intensity based on distance
            if dist < 2:
                canvas[i][j] = '█'
            elif dist < 3.5:
                canvas[i][j] = '▓'
            elif dist < 5:
                canvas[i][j] = '▒'
            elif dist < 6.5:
                canvas[i][j] = '░'

# LEFT: True coupling - closed loop (∇ flows through ∇)
paint_flow(canvas, W//4, H//2, closed=True)

# RIGHT: Broken coupling - severed (∇ flows to void)
paint_flow(canvas, 3*W//4, H//2, closed=False)

# Print the painting
for row in canvas:
    print(''.join(row))
